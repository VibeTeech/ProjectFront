<script>
  var pontuacao = 0;

  function verificarRespostas() {
    var respostas = document.getElementsByName("pergunta1");
    var respostaCorreta = false;

    for (var i = 0; i < respostas.length; i++) {
      if (respostas[i].checked && respostas[i].value === "correto") {
        respostaCorreta = true;
        break;
      }
    }

    if (respostaCorreta) {
      pontuacao++;
      document.getElementById("pontuacao").innerHTML = pontuacao;
      alert("Resposta correta! Sua pontuação aumentou para " + pontuacao);
    } else {
      alert("Resposta incorreta! Tente novamente.");
    }
  }
</script>

</body>
</html>


//EXEMPLO VALIDAÇÃO DA CONTA

function validarConta() {
    var conta = document.getElementById("conta").value;

    // Verifica se o campo está vazio
    if (conta === "") {
      document.getElementById("resultado").innerHTML = "Por favor, digite sua conta.";
      return;
    }

    // Verifica se o formato da conta é válido (por exemplo, se tem pelo menos 6 caracteres)
    if (conta.length < 6) {
      document.getElementById("resultado").innerHTML = "A conta deve ter pelo menos 6 caracteres.";
      return;
    }

    // Se passou pelas verificações, considera a conta como válida
    document.getElementById("resultado").innerHTML = "Conta válida: " + conta;
  }
</script>

function validateMath() {
    var answer = document.getElementById('answer').value;
    var resultElement = document.getElementById('result');
    var correctAnswer = 7; // A resposta correta para 3 + 4

    if (parseInt(answer) === correctAnswer) {
        resultElement.textContent = 'Resposta correta!';
        resultElement.style.color = 'green';

        // Sublinhando a opção correta
        document.getElementById('option2').classList.add('correct');
    } else {
        resultElement.textContent = 'Resposta incorreta. Tente novamente.';
        resultElement.style.color = 'red';
    }
}
